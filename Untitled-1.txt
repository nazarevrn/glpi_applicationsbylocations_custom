версии и их id по имени ПО

SELECT
	s.id as softId,
	s.name as softName,
	sv.id as versionId,
	sv.name as versionName
FROM 
     glpi_softwares s
LEFT JOIN
	glpi_softwareversions sv ON sv.softwares_id = s.id
WHERE s.name = 'Autodesk 360'

что-то похожее на нужный запрос

SELECT
	pc.users_id as userId,
	us.name as userName,
	pc.name as pcName,
	l.name
FROM 
	glpi_computers pc
LEFT JOIN 
    glpi_users us ON us.id = pc.users_id AND us.is_active = 1
LEFT JOIN
	glpi_locations l ON l.id = us.locations_id
WHERE
	pc.id IN (
		SELECT 
			pc_ver.computers_id
		FROM 
			glpi_computers_softwareversions pc_ver
		WHERE
			pc_ver.softwareversions_id 	IN (
				SELECT
					sv.id as versionId
				FROM 
					glpi_softwares s
				LEFT JOIN
					glpi_softwareversions sv ON sv.softwares_id = s.id
				WHERE s.name LIKE '%JetBrains%'
				)
		AND
		pc_ver.is_deleted_computer != 1)
